\begin{scnrelfromlist}{аргументы агента}
    \scnitem{question\_instance}
    \begin{scnindent}
        \scnxplanation{действие, которое необходимо протранслировать, у которого есть ответ связанный отношением nrel\_answer}
        \scntext{отношение аргумента}{исходная sc-конструкция'}
    \end{scnindent}
    \scnitem{format}
    \begin{scnindent}
        \scnxplanation{формат в которой необходимо протранслировать, в данном случае формат scn (json)}
        \scntext{отношение аргумента}{выходной формат'}
    \end{scnindent}
    \scnitem{lang}
    \begin{scnindent}
        \scnxplanation{язык содержимого sc-ссылок и идентификаторов}
        \scntext{отношение аргумента}{ui\_rrel\_user\_lang}
    \end{scnindent}
\end{scnrelfromlist}
\begin{scnrelfromlist}{необязательные аргументы агента}
    \scnitem{filter\_set}
    \begin{scnindent}
        \scnxplanation{множество фильтрации (множество отношений, которые не нужно включать в результаты трансляции)}
        \scntext{отношение аргумента}{ui\_rrel\_filter\_list}
    \end{scnindent}
\end{scnrelfromlist}
